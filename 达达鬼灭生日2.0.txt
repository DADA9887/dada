function modifyDateByLabel(labelText, newDate) { 
    let labels = document.querySelectorAll('.block-mypage-member-info-label'); 
    for (let label of labels) {
         if (label.innerText.trim() === labelText) {
             let ddElement = label.closest('dl').querySelector('.block-mypage-member-info-value'); 
            if (ddElement) {
                 ddElement.innerText = newDate;
                 return 'dadaNB'; } } } 
    return 'Not Find'; } 

function modifyDateByLabel2(labelText, newDate) { 
    let labels = document.querySelectorAll('.block-mypage-history-block-detail-contents-block-label'); 
    for (let label of labels) {
         if (label.innerText.trim() === labelText) {
             let ddElement = label.closest('dl').querySelector('.block-mypage-history-block-detail-contents-block-content'); 
            if (ddElement) {
                 ddElement.innerText = newDate;
                 return 'dadaNB'; } } } 
    return 'Not Find'; } 

modifyDateByLabel('生年月日', data);
